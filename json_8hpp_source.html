1584  {
1585  // check if each element is an array with two elements whose first
1586  // element is a string
1587  bool is_an_object = std::all_of(init.begin(), init.end(),
1588  [](const basic_json & element)
1589  {
1590  return element.is_array() and element.size() == 2 and element[0].is_string();
1591  });
1592 
1593  // adjust type if type deduction is not wanted
1594  if (not type_deduction)
1595  {
1596  // if array is wanted, do not create an object though possible
1597  if (manual_type == value_t::array)
1598  {
1599  is_an_object = false;
1600  }

1602  // if object is wanted but impossible, throw an exception
1603  if (manual_type == value_t::object and not is_an_object)
1604  {
1605  throw std::domain_error("cannot create object from initializer list");

7278  static string_t to_unicode(const std::size_t codepoint1,
7279  const std::size_t codepoint2 = 0)
7280  {

7282  std::size_t codepoint = codepoint1;
7283 
7284  // check if codepoint1 is a high surrogate
7285  if (codepoint1 >= 0xD800 and codepoint1 <= 0xDBFF)
7286  {
7287  // check if codepoint2 is a low surrogate
7288  if (codepoint2 >= 0xDC00 and codepoint2 <= 0xDFFF)
7289  {
7290  codepoint =
7291  // high surrogate occupies the most significant 22 bits
7292  (codepoint1 << 10)
7293  // low surrogate occupies the least significant 15 bits
7294  + codepoint2
7295  // there is still the 0xD800, 0xDC00 and 0x10000 noise
7296  // in the result so we have to subtract with:
7297  // (0xD800 << 10) + DC00 - 0x10000 = 0x35FDC00
